n=int(input())
c=0
if(n>=5):
    c+=n//5
    n=n%5
if(n>=4):
    c+=n//4
    n=n%4
if(n>=3):
    c+=n//3
    n=n%3
if(n>=2):
    c+=n//2
    n=n%2 
else:
    c+=n

print(c)
