n=int(input())
n_s=n//2
n_d=1
for i in range(n):
    s="*"*n_s +"D"*n_d+"*"*n_s
    print(s)
    if(i<n//2):
        n_s-=1
        n_d+=2
    elif (i>=n//2):   
        n_s+=1
        n_d-=2
